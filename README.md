# Linux Command
# 一般常用  
sudo ....   最高權限 (ex: sudo lsusb, sudo python hello.py...)  
cd ..		    上一層  
cd usr/pi/	進入pi資料夾  
ls 		      目前檔案清單  
clear 清除cmd  
  
# 使用Nano開啟/編譯
nano hello.py	  開啟 hello.py (如果沒有檔案, 自動產生新檔)  
python hello.py 編譯並執行 hello.py  
  
# 列出使用的 usb/port 清單  
lsusb       usb	清單  
ls -l /dev/tty* 列出所有使用的port (控Arduino常用)  
  
# Ipconfig  
ip a  
ip addr  
ip address  
ip addr show
